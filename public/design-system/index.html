Tokens CSS

Variáveis CSS do projeto. Declare em :root em cada componente ou num arquivo global.

/* ── Cores ── */ :root { --color-black: #242320; /* foreground principal */ --color-soft: #5f5b57; /* texto secundário */ --color-muted: #78736f; /* texto terciário */ --color-placeholder: #9e9a97; /* labels, placeholders */ --color-border: #f0efed; /* separadores */ --color-surface: #edeceb; /* superfície elevada */ --color-bg: #f7f7f5; /* fundo do app */ --color-white: #ffffff; /* cards e inputs */ --color-danger: #b94040; /* ações destrutivas */ --color-danger-bg: #fce8e8; /* bg de ações destrutivas */ /* ── Border Radius ── */ --radius-sm: 8px; --radius-md: 12px; --radius-lg: 16px; --radius-full: 9999px; /* ── Sombras ── */ --shadow-icon: 0px 5px 20px rgba(95,91,87,0.08), 0px 4px 8px rgba(95,91,87,0.10), 0px 2px 2px rgba(95,91,87,0.10); --shadow-card: 0px 2px 8px rgba(95,91,87,0.06); --shadow-menu: 0 8px 24px rgba(0,0,0,0.10), 0 2px 8px rgba(0,0,0,0.06); --shadow-toast: 0px 8px 24px rgba(0,0,0,0.10), 0px 2px 6px rgba(0,0,0,0.06); }
